# STDD Skill: /stdd:mock

## Purpose
**为外部依赖生成测试 Mock、Stub 与 Fake**。这是 Chaos Code 的 Mock 生成 skill，为测试创建测试替身。

**核心设计原则：**
- **语言无关**：支持多种 Mock 模式
- **接口驱动**：从 spec/schema 生成
- **安全默认**：安全的默认返回值
- **调用历史**：可验证调用记录

## When to Use
- 需要隔离外部依赖时
- 需要模拟 API 响应时
- 需要控制测试场景时
- 需要加速测试时

## Test Double 类型

### Mock（模拟对象）
**用途**：验证行为
- 记录调用
- 验证调用次数
- 验证调用参数

### Stub（存根）
**用途**：提供测试数据
- 返回预设值
- 模拟特定场景
- 不关心调用

### Fake（伪造对象）
**用途**：简化实现
- 内存实现
- 简化的数据库
- 不依赖外部服务

## CLI Runtime

```bash
# 生成 Mock
chaos mock <change-id>

# 指定接口
chaos mock <change-id> --interface UserService

# 指定类型
chaos mock <change-id> --type stub
chaos mock <change-id> --type fake

# 从 API spec 生成
chaos mock <change-id> --from-api-spec

# Workspace 支持
chaos mock <change-id> --workspace packages/api
```

## 多语言 Mock 示例

### JavaScript/TypeScript

#### Mock (Sinon.js)
```typescript
// mocks/UserService.mock.ts
import { sinon } from 'sinon';

export const UserServiceMock = {
  findByUsername: sinon.stub().resolves({
    id: '1',
    username: 'test',
    email: 'test@example.com'
  }),
  
  create: sinon.stub().resolves({
    id: '2',
    username: 'newuser'
  })
};

// 验证调用
expect(UserServiceMock.findByUsername.calledWith('test')).toBe(true);
```

#### Stub (手动)
```typescript
// stubs/ApiClient.stub.ts
export const ApiClientStub = {
  get: async (url: string) => ({
    status: 200,
    data: { message: 'OK' }
  }),
  
  post: async (url: string, data: any) => ({
    status: 201,
    data: { id: '123', ...data }
  })
};
```

### Java

#### Mock (Mockito)
```java
// mocks/UserServiceMock.java
import org.mockito.Mockito;
import static org.mockito.Mockito.*;

public class UserServiceMock {
    public static UserService create() {
        UserService mock = Mockito.mock(UserService.class);
        
        when(mock.findByUsername("test"))
            .thenReturn(Optional.of(new User("1", "test", "test@example.com")));
        
        when(mock.create(any()))
            .thenReturn(new User("2", "newuser", "new@example.com"));
        
        return mock;
    }
}

// 验证调用
verify(userService, times(1)).findByUsername("test");
```

### Go

#### Mock (gomock + mockgen)
```go
// mocks/user_service_mock.go
// Code generated by mockgen; DO NOT EDIT.

package mocks

import "github.com/golang/mock/gomock"

type MockUserService struct {
    ctrl     *gomock.Controller
    recorder *MockUserServiceMockRecorder
}

func (m *MockUserService) FindByUsername(username string) (*User, error) {
    m.ctrl.T.Helper()
    ret := m.ctrl.Call(m, "FindByUsername", username)
    ret0, _ := ret[0].(*User)
    ret1, _ := ret[1].(error)
    return ret0, ret1
}
```

#### Stub (手动)
```go
// stubs/api_client_stub.go
type ApiClientStub struct {
    Responses map[string]interface{}
}

func (s *ApiClientStub) Get(url string) (map[string]interface{}, error) {
    if resp, ok := s.Responses[url]; ok {
        return resp.(map[string]interface{}), nil
    }
    return map[string]interface{}{"status": 200, "data": map[string]string{"message": "OK"}}, nil
}
```

### Rust

#### Mock (mockall)
```rust
// mocks/user_service_mock.rs
use mockall::{mock, predicate::*};

mock! {
    UserService {}

    impl UserService for UserService {
        fn find_by_username(&self, username: &str) -> Option<User>;
        fn create(&self, user: User) -> Result<User, Error>;
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    
    #[test]
    fn test_find_user() {
        let mut mock = MockUserService::new();
        mock.expect_find_by_username()
            .with(eq("test"))
            .returning(|_| Some(User::new("1", "test", "test@example.com")));
        
        let user = mock.find_by_username("test").unwrap();
        assert_eq!(user.username, "test");
    }
}
```

### C#

#### Mock (Moq)
```csharp
// mocks/UserServiceMock.cs
using Moq;

public class UserServiceMock
{
    public static Mock<IUserService> Create()
    {
        var mock = new Mock<IUserService>();
        
        mock.Setup(x => x.FindByUsername("test"))
            .ReturnsAsync(new User { Id = "1", Username = "test", Email = "test@example.com" });

        return mock;
    }
}

// 验证调用

```

## HTTP/API Mocking

### MSW (Mock Service Worker) - JavaScript/TypeScript
```typescript
// msw/handlers.ts
import { http, HttpResponse } from 'msw';

export const handlers = [
  // 成功响应
  http.get('/api/users/:id', ({ params }) => {
    return HttpResponse.json({
      id: params.id,
      name: 'Test User',
      email: 'test@example.com'
    });
  }),

  // 错误响应
  http.post('/api/users', async ({ request }) => {
    const body = await request.json();
    
    if (!body.email) {
      return HttpResponse.json(
        { error: 'Email is required' },
        { status: 400 }
      );
    }
    
    return HttpResponse.json(
      { id: '123', ...body },
      { status: 201 }
    );
  }),

  // 超时模拟
  http.get('/api/slow', () => {
    return new Promise(() => {}); // 永不返回
  }),

  // 网络错误
  http.get('/api/error', () => {
    return HttpResponse.error();
  })
];

// msw/server.ts
import { setupServer } from 'msw/node';
import { handlers } from './handlers';

export const server = setupServer(...handlers);

// 测试设置
beforeAll(() => server.listen());
afterEach(() => server.resetHandlers());
afterAll(() => server.close());
```

### Nock - HTTP Mocking for Node.js
```javascript
// nock/httpMock.js
const nock = require('nock');

describe('API Client', () => {
  beforeEach(() => {
    nock.cleanAll();
  });

  it('should fetch user', async () => {
    nock('https://api.example.com')
      .get('/users/1')
      .reply(200, {
        id: '1',
        name: 'Test User'
      });

    const user = await apiClient.getUser('1');
    expect(user.name).toBe('Test User');
  });

  it('should handle errors', async () => {
    nock('https://api.example.com')
      .get('/users/999')
      .reply(404, {
        error: 'User not found'
      });

    await expect(apiClient.getUser('999')).rejects.toThrow('User not found');
  });
});
```

### WebMock - Ruby
```ruby
# spec/support/webmock.rb
require 'webmock/rspec'

    expect(user['name']).to eq('Test User')
  end
end
```

## 设计决策

### 为什么需要 Mock？
- **隔离**: 隔离外部依赖
- **速度**: 快速执行测试
- **控制**: 控制测试场景

### 为什么区分 Mock/Stub/Fake？
- **清晰**: 明确用途
- **选择**: 选择合适的工具
- **维护**: 更好的可维护性

### 为什么 MSW 模式？
- **真实性**: 拦截真实网络请求
- **复用**: 同一 mocks 用于测试和开发
- **类型安全**: TypeScript 完全支持

### 为什么 VCR 模式？
- **真实数据**: 记录真实 API 响应
- **简单**: 无需手动编写 mocks
- **同步**: 自动与 API 变更同步
